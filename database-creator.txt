CREATE TABLE routes 
    (route_id Integer Primary key, 
    areas Varchar(256) NOT NULL,
    name Varchar(100) NOT NULL,
    yds Varchar(20) NOT NULL,
    type Varchar(100) NOT NULL,
    fa Varchar(100) NOT NULL,
    page_views Varchar(100) NOT NULL,
    description Varchar(2000) NOT NULL,
    location Varchar(2000) NOT NULL,
    protection Varchar(2000) NOT NULL,
    link Varchar(100) NOT NULL);

CREATE TABLE suggested_ratings
    (user_id Integer Primary key,
    suggested_rating Varchar(20) NOT NULL);
    
CREATE TABLE comments
    (comment_id Integer Primary key,
    route_id Integer NOT NULL,
    user_id Integer NOT NULL,
    comment Varchar(2000) NOT NULL);
